# Menu

## Usage

```tsx
import { Menu } from '@divriots/starter-tailwindcss/menu';
import { PencilIcon, DuplicateIcon } from '@heroicons/react/solid';

<Menu
  items={[
    {
      Icon: PencilIcon,
      label: 'Edit',
      onClick: () => window.alert('Edit was pressed!'),
    },
    { Icon: DuplicateIcon, label: 'Duplicate' },
    { label: 'Exit' },
  ]}
/>;
```
