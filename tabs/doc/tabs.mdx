## Usage

```tsx
import { Tabs } from '@divriots/starter-tailwindcss/tabs';

<Tabs
  items={[
    {
      label: 'Recent',
      panel: () => ({
        /** ReactNode */
      }),
    },
    {
      label: 'Popular',
      panel: () => ({
        /** ReactNode */
      }),
    },
    {
      label: 'Trending',
      panel: () => ({
        /** ReactNode */
      }),
    },
  ]}
/>;
```
